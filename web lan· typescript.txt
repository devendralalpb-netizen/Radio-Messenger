import React, { useState, useEffect } from "react";

export default function RadioMessenger() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState("");

  // Simulated radio broadcast using localStorage (acts like LAN broadcast)
  useEffect(() => {
    const interval = setInterval(() => {
      const data = localStorage.getItem("radio_messages");
      if (data) {
        setMessages(JSON.parse(data));
      }
    }, 500);
    return () => clearInterval(interval);
  }, []);

  const sendMessage = () => {
    if (!input.trim()) return;

    const newMessage = {
      text: input,
      time: new Date().toLocaleTimeString(),
    };

    const updated = [...messages, newMessage];
    setMessages(updated);
    localStorage.setItem("radio_messages", JSON.stringify(updated));
    setInput("");
  };

  return (
    <div className="h-screen flex flex-col items-center justify-center bg-gray-900 text-white p-4">
      <h1 className="text-2xl font-bold mb-4">📡 Radio Messenger</h1>

      <div className="w-full max-w-md h-80 overflow-y-auto bg-gray-800 rounded-2xl p-4 mb-4 shadow-lg">
        {messages.map((msg, index) => (
          <div key={index} className="mb-2">
            <span className="text-green-400">[{msg.time}]</span> {msg.text}
          </div>
        ))}
      </div>

      <div className="flex w-full max-w-md">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          className="flex-1 p-2 rounded-l-xl bg-gray-700 outline-none"
          placeholder="Type message..."
        />
        <button
          onClick={sendMessage}
          className="bg-green-500 px-4 rounded-r-xl hover:bg-green-600"
        >
          Send
        </button>
      </div>
    </div>
  );
}
